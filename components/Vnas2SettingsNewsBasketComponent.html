import { Component, OnInit, Injector } from '@angular/core';
import { BaseComponent } from '@vnas2-workspace/vnas2-core';
import { ActivatedRoute } from '@angular/router';
import { Vnas2SettingsSandboxService } from '../../vnas2-settings.sandbox';
import { PagesModel } from '@vnas2-workspace/vnas2-ui-commons';

@Component({
  selector: 'app-vnas2-settings-news-basket',
  templateUrl: './vnas2-settings-news-basket.component.html',
  styleUrls: ['./vnas2-settings-news-basket.component.scss']
})
export class Vnas2SettingsNewsBasketComponent extends BaseComponent
  implements OnInit {
  objectType: string;
  constructor(
    injector: Injector,
    private activatedRoute: ActivatedRoute,
    private vnas2SettingsSandboxService: Vnas2SettingsSandboxService
  ) {
    super(injector);
  }

  ngOnInit() {
    const pages = this.activatedRoute.snapshot.data as PagesModel;
    this.titlePage = pages.title;
    this.objectType = pages.type;

    this.vnas2SettingsSandboxService.setTitle(this.titlePage);

    this.setTitle(this.titlePage);
  }
}
