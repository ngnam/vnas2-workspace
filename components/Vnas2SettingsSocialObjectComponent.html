import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Component, Injector, OnDestroy, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import {
  SocialObjectService,
  SocialObject,
  SocialObjectDTO
} from '@vnas2-workspace/vnas2-api';
import {
  BaseComponent,
  isNotEmpty,
  Logger,
  AppConfig,
  SnackBarService
} from '@vnas2-workspace/vnas2-core';
import { Observable, of, Subject } from 'rxjs';
import { catchError, map, shareReplay, takeUntil } from 'rxjs/operators';
import { Vnas2SettingsSandboxService } from '../../vnas2-settings.sandbox';
import { PagesModel } from '@vnas2-workspace/vnas2-ui-commons';
import { IAppDialogService } from '@vnas2-workspace/vnas2-ui-commons';
import { AppSettingsConfirmComponent } from '../../components/app-settings-confirm/app-settings-confirm.component';

const logger = new Logger('Vnas2SettingsSocialObjectComponent');

@Component({
  selector: 'app-vnas2-settings-social-object',
  templateUrl: './vnas2-settings-social-object.component.html',
  styleUrls: ['./vnas2-settings-social-object.component.scss']
})
export class Vnas2SettingsSocialObjectComponent extends BaseComponent
  implements OnInit, OnDestroy {
  socialObjects$: Observable<any>;
  socialObjects: any;

  isMobile$: Observable<boolean> = this.breakpointObserver
    .observe([Breakpoints.XSmall, Breakpoints.Small])
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  filterSocialObjects: {
    page_size?: string;
    page_number?: string;
  } = {
    page_size: null,
    page_number: null
  };

  private destroyed$ = new Subject();
  constructor(
    injector: Injector,
    private activatedRoute: ActivatedRoute,
    private breakpointObserver: BreakpointObserver,
    private vnas2SettingsSandboxService: Vnas2SettingsSandboxService,
    private socialObjectService: SocialObjectService,
    private appDialogService: IAppDialogService,
    private notificationService: SnackBarService
  ) {
    super(injector);
    this.filterSocialObjects.page_size = this.pageSize;
    this.filterSocialObjects.page_number = this.pageNumber;
  }

  ngOnInit() {
    this.updatePageTitle();
    this.loadSocialObjects(this.filterSocialObjects);
  }

  ngOnDestroy() {
    this.destroyed$.next();
    this.destroyed$.complete();
  }
  //#region public method

  onSubmitForm(event: {
    data: SocialObjectDTO;
    type: string;
    socialObjectId?: number;
  }) {
    const { data, type, socialObjectId } = event;
    if (!data || !type) {
      return;
    }
    data.status = +data.status;
    data.object_type = +data.object_type;
    switch (type) {
      case AppConfig.ActionType.INSERT:
        const insert$ = this.createSettings(data);
        insert$.subscribe(response => {
          if (response) {
            logger.debug(response);
            this.loadSocialObjects(this.filterSocialObjects);
          }
        });
        break;

      case AppConfig.ActionType.UPDATE:
        const update$ = this.updateSettings(socialObjectId, data);
        update$.subscribe(response => {
          if (response) {
            logger.debug(response);
            this.loadSocialObjects(this.filterSocialObjects);
          }
        });
        break;
    }
  }

  onGoToPageSocialObject(event: any) {
    this.filterSocialObjects.page_number = event;
    this.loadSocialObjects(this.filterSocialObjects);
  }

  onSearchSocialObject(query: string) {
    this.searchText(query);
  }

  onConfirmDeleteObject(object?: SocialObject) {
    logger.debug(object);
    if (!object) {
      return;
    }
    const dialogRef = this.appDialogService.openConfirmYesNoDialog(
      `Xác nhận xóa ${object.account_name}`,
      AppSettingsConfirmComponent,
      AppConfig.MESSAGE.ACCEPT,
      AppConfig.MESSAGE.CANCEL,
      AppConfig.MESSAGE.CONFIRM
    );
    dialogRef.subscribe(result => {
      if (result) {
        this.removeSettings(object.social_obj_id).subscribe(res => {
          logger.debug(res);
          if (res) {
            this.loadSocialObjects(this.filterSocialObjects);
          }
        });
      }
    });
  }
  //#endregion

  private loadSocialObjects(filter: {
    page_size?: string;
    page_number?: string;
  }) {
    this.socialObjects$ = this.socialObjectService
      .loadSocialObjectAdvancedSearch(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.socialObjects$.subscribe(result => (this.socialObjects = result));
  }

  private createSettings(data: SocialObjectDTO) {
    return this.socialObjectService.createSocialObject(data).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} <br> ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private updateSettings(id: number, data: SocialObjectDTO) {
    return this.socialObjectService.updateSocialObject(id, data).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} <br> ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private removeSettings(id: number) {
    return this.socialObjectService.removeSocialObject(id).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} \n ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  //#region private method
  private updatePageTitle() {
    const pages = this.activatedRoute.snapshot.data as PagesModel;
    this.titlePage = `${pages.title}`;
    this.vnas2SettingsSandboxService.setTitle(`Cấu hình ${this.titlePage}`);
    this.setTitle(`Cấu hình ${this.titlePage}`);
  }

  private searchText(query: string) {
    if (isNotEmpty(query)) {
      query = query.trim();
      // call api search in here
    }
  }

  private catchError(err) {
    logger.debug(err);
    return of(null);
  }
  //#endregion
}
