import {
  Component,
  OnInit,
  Output,
  EventEmitter,
  Input,
  OnChanges,
  OnDestroy
} from '@angular/core';
import { FormGroup, FormBuilder } from '@angular/forms';
import { isNullOrUndefined } from 'util';
import { fadeInAnimation } from '../../animations';
import { Observable, Subscription } from 'rxjs';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { map, shareReplay } from 'rxjs/operators';

@Component({
  selector: 'app-form-search-news-advanced',
  templateUrl: './form-search-news-advanced.component.html',
  styleUrls: ['./form-search-news-advanced.component.scss'],
  animations: [fadeInAnimation]
})
export class FormSearchNewsAdvancedComponent
  implements OnInit, OnChanges, OnDestroy {
  formSearch: FormGroup;
  @Output() searchSubmit: EventEmitter<any> = new EventEmitter();
  @Input() languages: { name: string; value: string }[] = null;
  @Input() websites: { website_cate_id: number; website_url: string }[] = null;

  ratingPoints = Array.from(Array(11).keys());
  floatLabel = 'auto';
  private subcription: Subscription = new Subscription();
  constructor(
    private formBuilder: FormBuilder,
    private breakpointObserver: BreakpointObserver
  ) {
    this.ratingPoints.unshift(-1);
  }

  isToggleResultSearch = false;
  isConditionsMore = false;

  languagesDictionaries = {};
  websitesDictionaries = {};

  isMobile$: Observable<boolean> = this.breakpointObserver
    .observe([Breakpoints.XSmall, Breakpoints.Small])
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  ngOnInit() {
    this.initializedForm();
  }

  ngOnChanges() {
    if (this.languages && this.languagesDictionaries) {
      this.languages.forEach(lang => {
        this.languagesDictionaries[lang.value] = lang.name;
      });
    }
    if (this.websites && this.websitesDictionaries) {
      this.websites.forEach(item => {
        this.websitesDictionaries[item.website_cate_id] = item.website_url;
      });
    }
    if (this.formSearch) {
      this.subcription.add(
        this.formSearch.valueChanges.subscribe(
          () => (this.isToggleResultSearch = true)
        )
      );
    }
  }

  ngOnDestroy() {
    if (this.subcription) {
      this.subcription.unsubscribe();
    }
  }

  private initializedForm() {
    this.formSearch = this.formBuilder.group({
      fromDateSearch: [new Date(2000, 0, 1), null],
      toDateSearch: [null, null],
      language: [[], null], // multiple select
      website_cate_id: [[], null],
      rating_point: ['', null]
    });
  }

  onSearch() {
    if (
      this.formSearch.valid &&
      (!isNullOrUndefined(this.formSearch.get('fromDateSearch').value) ||
        !isNullOrUndefined(this.formSearch.get('toDateSearch').value))
    ) {
      this.searchSubmit.emit({ data: this.formSearch.value });
    }
  }

  refreshData(control: number, event?: any) {
    event.stopPropagation();
    let controlName = '';
    switch (control) {
      case 1:
        controlName = 'fromDateSearch';
        break;
      case 2:
        controlName = 'toDateSearch';
        break;
      case 3:
        controlName = 'language';
        break;
      case 4:
        controlName = 'website_cate_id';
        break;
      case 5:
        controlName = 'rating_point';
        break;
    }

    setTimeout(() => {
      if (this.formSearch.get(controlName)) {
        this.formSearch.get(controlName).setValue('');
      }
    }, 200);
  }
}
