import { Component, OnInit, Input, EventEmitter, Output, OnChanges } from '@angular/core';
import { ObjectPerson } from '../../models/object-person';

@Component({
  selector: 'app-list-objects',
  templateUrl: './list-objects.component.html',
  styleUrls: ['./list-objects.component.scss']
})
export class ListObjectsComponent implements OnChanges {
  @Input() listObjects: ObjectPerson[] = null;
  @Output() clickObject: EventEmitter<any> = new EventEmitter<any>();
  dataSource = {
    data: this.listObjects
  };
  constructor() {
  }

  ngOnChanges() {
    this.dataSource.data = null;
    if (this.listObjects) {
    setTimeout(() => {
      this.dataSource.data = this.listObjects;
    }, 600);
    }

  }

  onClickObject($event: ObjectPerson) {
    this.clickObject.emit({ objectPerson: $event });
  }
}
