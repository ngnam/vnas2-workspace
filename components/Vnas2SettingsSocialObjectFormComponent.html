import {
  Component,
  EventEmitter,
  Input,
  OnChanges,
  OnInit,
  Output,
  ViewChild
} from '@angular/core';
import { FormBuilder, FormGroup, NgForm, Validators } from '@angular/forms';
import { SocialObject, SocialObjectDTO } from '@vnas2-workspace/vnas2-api';
import { AppConfig } from '@vnas2-workspace/vnas2-core';
import { compareTwoFormControl } from '@vnas2-workspace/vnas2-core';
@Component({
  selector: 'app-vnas2-settings-social-object-form',
  templateUrl: './vnas2-settings-social-object-form.component.html',
  styleUrls: ['./vnas2-settings-social-object-form.component.scss']
})
export class Vnas2SettingsSocialObjectFormComponent
  implements OnInit, OnChanges {
  formSocialObject: FormGroup;
  title = 'Taọ mới';
  floatLabel: 'never';
  btnSubmitText = 'Tạo mới';
  type = AppConfig.ActionType.INSERT;
  private socialObject: SocialObject = null;
  listOfStatus: { name: string; value: number }[] = AppConfig.ListOfStatus;
  @Input() titleForm: string;
  @Output() submitForm: EventEmitter<any> = new EventEmitter<any>();
  @Output() deleteData: EventEmitter<any> = new EventEmitter<any>();
  @ViewChild('formDirective', { static: true }) private formDirective: NgForm;

  constructor(private formBuilder: FormBuilder) {}

  ngOnInit() {
    this.initializeForm();
  }
  ngOnChanges(): void {
    if (this.titleForm) {
      this.title = `Tạo mới ${this.titleForm}`;
    }
  }

  onConfirmDelete() {
    if (!this.socialObject) {
      return;
    }
    this.deleteData.emit({ data: this.socialObject });
  }

  onSubmit() {
    if (this.formSocialObject.valid) {
      const socialObject: SocialObjectDTO = {
        account_name: this.formSocialObject.get('account_name').value,
        account_password: this.formSocialObject.get('account_password').value,
        face_id: this.formSocialObject.get('face_id').value,
        object_type: this.formSocialObject.get('object_type').value,
        other_information: this.formSocialObject.get('other_information').value,
        social_name: this.formSocialObject.get('social_name').value,
        social_network: this.formSocialObject.get('social_network').value,
        status: this.formSocialObject.get('status').value
      };
      this.submitForm.emit({
        data: socialObject,
        type: this.type,
        socialObjectId: this.socialObject
          ? this.socialObject.social_obj_id
          : null
      });
    }
  }

  resetForm() {
    this.type = AppConfig.ActionType.INSERT;
    this.title = `Tạo mới ${this.titleForm}`;
    this.formSocialObject.reset();
    this.formDirective.resetForm();
    this.socialObject = null;
  }

  setValueForm(object?: SocialObject) {
    if (!object) {
      return;
    }
    this.socialObject = object;
    this.type = AppConfig.ActionType.UPDATE;
    this.title = `Cập nhật ${object.account_name}`;
    this.btnSubmitText = 'Cập nhập';

    this.formSocialObject.patchValue({
      account_name: object.account_name,
      face_id: object.face_id,
      object_type: 1,
      other_information: object.other_information,
      social_name: object.social_name,
      social_network: object.social_network,
      status: object.status
    });
  }

  private initializeForm() {
    this.formSocialObject = this.formBuilder.group(
      {
        account_name: ['', Validators.required],
        account_password: [
          '',
          [
            Validators.required,
            Validators.pattern(
              '/^(?=.*d)(?=.*[a-z])(?=.*[A-Z])[0-9a-zA-Z]{8,}$//^(?=.*d)(?=.*[a-z])(?=.*[A-Z])[0-9a-zA-Z]{8,}$'
            )
          ]
        ],
        confirm_password: ['', [Validators.required]],
        face_id: ['', Validators.required],
        object_type: [1, Validators.required],
        other_information: ['', Validators.required],
        social_name: ['', Validators.required],
        social_network: ['', Validators.required],
        status: ['', Validators.required]
      },
      {
        validators: compareTwoFormControl(
          'account_password',
          'confirm_password'
        )
      }
    );
  }
}
