import {
  Component,
  Input,
  EventEmitter,
  Output,
  OnChanges
} from '@angular/core';
import { NewBasket } from '../../models';

@Component({
  selector: 'app-news-basket',
  templateUrl: './news-basket.component.html',
  styleUrls: ['./news-basket.component.scss']
})
export class NewsBasketComponent implements OnChanges {
  @Input() newsBasket: NewBasket[] = null;
  @Output() clickNewBasket: EventEmitter<any> = new EventEmitter<any>();
  constructor() {}

  dataSource = {
    data: this.newsBasket
  };

  ngOnChanges() {
    this.dataSource.data = null;
    if (this.newsBasket) {
      setTimeout(() => {
        this.dataSource.data = this.newsBasket;
      }, 600);
    }
  }

  onClickNewBasket($event: NewBasket) {
    this.clickNewBasket.emit({ newBasket: $event });
  }
}
