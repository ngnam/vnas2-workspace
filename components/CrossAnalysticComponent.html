import { Component, OnInit, Injector } from '@angular/core';
import { BaseComponent } from '@vnas2-workspace/vnas2-core';
import { ActivatedRoute } from '@angular/router';
import { Vnas2ObjectsPageSandboxService } from '../../vnas2-object-page.sandbox';
import { PagesModel } from '@vnas2-workspace/vnas2-ui-commons';

@Component({
  selector: 'app-cross-analystic',
  templateUrl: './cross-analystic.component.html',
  styleUrls: ['./cross-analystic.component.scss']
})
export class CrossAnalysticComponent extends BaseComponent implements OnInit {
  // code here
  constructor(
    injector: Injector,
    private activatedRoute: ActivatedRoute,
    private vnas2ObjectsPageSandboxService: Vnas2ObjectsPageSandboxService
  ) {
    super(injector);
  }

  ngOnInit() {
    const pages = this.activatedRoute.snapshot.data as PagesModel;
    this.titlePage = pages.title;
    this.vnas2ObjectsPageSandboxService.setTitle(this.titlePage);
    this.setTitle(this.titlePage);
  }
}
