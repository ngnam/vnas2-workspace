import {
  Component,
  OnInit,
  OnDestroy,
  ViewChild,
  ViewContainerRef,
  ComponentRef,
  ComponentFactoryResolver,
  Inject
} from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material';
import { DialogData } from './app-dialog.model';

@Component({
  selector: 'ngn-app-dialog',
  templateUrl: './app-dialog.component.html',
  styleUrls: ['./app-dialog.component.scss']
})
export class AppDialogComponent implements OnInit, OnDestroy {
  title: string;
  isConfirm = false;
  buttonYesText = 'Ok';
  buttonNoText = 'No';
  messageConfirm: string;
  @ViewChild('target', { read: ViewContainerRef, static: true })
  vcRef: ViewContainerRef;

  componentRef: ComponentRef<any>;

  constructor(
    public dialogRef: MatDialogRef<AppDialogComponent>,
    private resolver: ComponentFactoryResolver,
    @Inject(MAT_DIALOG_DATA) public data: DialogData
  ) {}

  ngOnInit() {
    const factory = this.resolver.resolveComponentFactory(this.data.component);

    this.title = this.data['title'];
    this.isConfirm = this.data['isConfirm'] || false;
    this.buttonYesText = this.data['buttonYesText'] || this.buttonYesText;
    this.buttonNoText = this.data['buttonNoText'] || this.buttonNoText;
    this.messageConfirm = this.data['messageConfirm'];
    this.componentRef = this.vcRef.createComponent(factory);
  }

  onNoClick(): void {
    this.dialogRef.close(false);
  }

  ngOnDestroy() {
    if (this.componentRef) {
      this.componentRef.destroy();
    }
  }
}
