import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import {
  AfterViewInit,
  Component,
  Injector,
  OnDestroy,
  OnInit,
  ViewChild
} from '@angular/core';
import {
  MonitorObjectsService,
  NewsBasketService,
  NewsDomainApiService,
  NewsService
} from '@vnas2-workspace/vnas2-api';
import {
  AppConfig,
  BaseComponent,
  isEmpty,
  Logger
} from '@vnas2-workspace/vnas2-core';
import {
  fadeInAnimation,
  TreeCategoryVerticalComponent
} from '@vnas2-workspace/vnas2-ui-commons';
import * as moment from 'moment';
import { Observable, of, Subject } from 'rxjs';
import {
  catchError,
  finalize,
  map,
  shareReplay,
  takeUntil
} from 'rxjs/operators';
const logger = new Logger('DashBoardComponent');
@Component({
  selector: 'app-dash-board',
  templateUrl: './dash-board.component.html',
  styleUrls: ['./dash-board.component.scss'],
  animations: [fadeInAnimation]
})
export class DashBoardComponent extends BaseComponent
  implements OnInit, OnDestroy, AfterViewInit {
  private destroyed$ = new Subject();

  isNewsLoading = false;

  filterSearch: {
    page_number?: string;
    page_size?: string;
    from_date?: string;
    to_date?: string;
    table_names: string;
    order_by?: string; // field desc/asc
  } = {
    page_number: null,
    page_size: null,
    from_date: null,
    to_date: null,
    table_names: null,
    order_by: null
  };

  filterSearchConditions: { field_name: string; field_value: string }[];

  filterNewsDaily: {
    page_number?: string;
    page_size?: string;
    title_order?: string; // or
    date_created_order?: string; // or
    rating_point_order?: string; // or
  } = {
    page_number: null,
    page_size: null,
    title_order: null, // or
    date_created_order: null, // or
    rating_point_order: null // or
  };

  filterListObjects: {
    page_number?: string;
    page_size?: string;
    object_type: string;
    object_name: string;
  } = {
    page_number: null,
    page_size: null,
    object_type: null,
    object_name: null
  };

  filterWebsite: {
    page_number?: string;
    page_size?: string;
    website_url?: string;
  } = {
    page_number: null,
    page_size: null,
    website_url: null
  };

  filterCategoriesOrg: {
    page_number?: string;
    page_size?: string;
    object_type: string;
    object_name: string;
  } = {
    page_number: null,
    page_size: null,
    object_type: null,
    object_name: null
  };

  filterNewsBasket: {
    page_number?: string;
    page_size?: string;
  } = {
    page_number: null,
    page_size: null
  };

  languages = AppConfig.Languages;

  trees$: Observable<any>;

  categoriesOrg$: Observable<any>;
  categoriesOrg: any = null;

  newsBasket$: Observable<any>;
  newsBasket: any = null;

  dailyNews$: Observable<any>;
  dailyNews: any = null;

  listObjects$: Observable<any>;
  listObjects: any;

  news$: Observable<any>;
  news: any = null;

  website$: Observable<any>;
  wesites: any = null;

  isMobile$: Observable<boolean> = this.breakpointObserver
    .observe([Breakpoints.XSmall, Breakpoints.Small])
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  @ViewChild('treeCategory', { static: true })
  treeCategoryCmp: TreeCategoryVerticalComponent;

  constructor(
    injector: Injector,
    private newsDomainApiService: NewsDomainApiService,
    private newsBasketService: NewsBasketService,
    private newsService: NewsService,
    private monitorObjectsService: MonitorObjectsService,
    private breakpointObserver: BreakpointObserver
  ) {
    super(injector);
    this.titlePage = 'Bản tin';
    this.filterSearchConditions = [];
    this.filterSearch.page_number = this.pageNumber;
    this.filterSearch.page_size = this.pageSize;
    // API GET TOP NEWS_OF_DOMAIN
    this.filterSearch.table_names =
      'news_of_domain_10|news_of_domain_11|news_of_domain_12|news_of_domain_15';

    this.filterNewsDaily.page_number = this.pageNumber;
    this.filterNewsDaily.page_size = this.pageSize;

    this.filterWebsite.page_number = this.pageNumber;
    this.filterWebsite.page_size = this.pageSize;

    this.filterListObjects.page_number = this.pageNumber;
    this.filterListObjects.page_size = this.pageSize;
    this.filterListObjects.object_type = AppConfig.ObjectTypes.PERSON;

    this.filterCategoriesOrg.page_number = this.pageNumber;
    this.filterCategoriesOrg.page_size = this.pageSize;
    this.filterCategoriesOrg.object_type = AppConfig.ObjectTypes.COUNTRY;

    this.filterNewsBasket.page_number = this.pageNumber;
    this.filterNewsBasket.page_size = this.pageSize;
  }

  // #region life cycle hook methods order
  ngOnInit() {
    this.setTitle();
    this.loadTreeCategory();
    this.loadCategoriesOrg(this.filterCategoriesOrg);
    this.loadNewsBasket(this.filterNewsBasket);
    this.loadDailyNews(this.filterNewsDaily);
    this.loadListObjects(this.filterListObjects);
    this.loadNewsWebsite(this.filterWebsite);
    this.loadNews(this.filterSearch, this.filterSearchConditions);
  }

  ngOnDestroy() {
    this.destroyed$.next();
    this.destroyed$.complete();
    this.removeClassPage('DashBoardComponent');
  }

  ngAfterViewInit() {
    this.addClassPage('DashBoardComponent');
  }
  //#endregion

  //#region public methods

  onSelectedNode(node: any) {
    const data = node.data;
    let domainTablesName: any[] = [];
    domainTablesName = data.map((item: any) => item.domain_table_name);
    this.filterSearch.table_names = domainTablesName.join('|');
    this.isNewsLoading = false;
    this.loadNews(this.filterSearch, this.filterSearchConditions);
  }

  onSearch(data: {
    fromDateSearch: any;
    toDateSearch: any;
    language: any;
    website_cate_id: any;
    rating_point: any;
  }) {
    if (!isEmpty(data.fromDateSearch)) {
      this.filterSearch.from_date = moment(data.fromDateSearch).format(
        'YYYY-MM-DD'
      );
    }
    if (!isEmpty(data.toDateSearch)) {
      this.filterSearch.to_date = moment(data.toDateSearch).format(
        'YYYY-MM-DD'
      );
    }

    this.filterSearchConditions = [];

    if (!isEmpty(data.rating_point)) {
      this.filterSearchConditions.push({
        field_name: 'rating_point',
        field_value: data.rating_point.toString()
      });
    }
    if (!isEmpty(data.website_cate_id)) {
      this.filterSearchConditions.push({
        field_name: 'website_cate_id',
        field_value: data.website_cate_id.join('|')
      });
    }
    if (data.language.length > 0) {
      this.filterSearchConditions.push({
        field_name: 'language',
        field_value: data.language.join('|')
      });
    }

    this.isNewsLoading = false;
    this.loadNews(this.filterSearch, this.filterSearchConditions);
  }

  sortOrderByNewDaily(orderBy: { [key: number]: boolean }) {
    logger.debug(orderBy[0], orderBy[1], orderBy[2]);
    // title
    if (orderBy[0] !== undefined) {
      this.filterNewsDaily.title_order = orderBy[0]
        ? AppConfig.SortType.ASC
        : AppConfig.SortType.DESC;
    } else {
      this.filterNewsDaily.title_order = null;
    }
    // date
    if (orderBy[1] !== undefined) {
      this.filterNewsDaily.date_created_order = orderBy[1]
        ? AppConfig.SortType.ASC
        : AppConfig.SortType.DESC;
    } else {
      this.filterNewsDaily.date_created_order = null;
    }
    // rating
    if (orderBy[2] !== undefined) {
      this.filterNewsDaily.rating_point_order = orderBy[2]
        ? AppConfig.SortType.ASC
        : AppConfig.SortType.DESC;
    } else {
      this.filterNewsDaily.rating_point_order = null;
    }

    this.loadDailyNews(this.filterNewsDaily);
  }

  onPagingWebsite(page) {
    this.filterWebsite.page_number = page;
    this.loadNewsWebsite(this.filterWebsite);
  }

  onPagingDailyNews(page) {
    this.filterNewsDaily.page_number = page;
    this.loadDailyNews(this.filterNewsDaily);
  }

  onPagingCategoriesOrg(page) {
    this.filterCategoriesOrg.page_number = page;
    this.loadCategoriesOrg(this.filterCategoriesOrg);
  }

  onPagingListObjects(page) {
    this.filterListObjects.page_number = page;
    this.loadListObjects(this.filterListObjects);
  }

  onPagingNewsBasket(page) {
    this.filterNewsBasket.page_number = page;
    this.loadNewsBasket(this.filterNewsBasket);
  }

  onSearchWebsite(query) {
    if (!isEmpty(query)) {
      query = query.trim();
      this.filterWebsite.website_url = query;
    } else {
      this.filterWebsite.website_url = null;
    }
    this.loadNewsWebsite(this.filterWebsite);
  }
  //#endregion

  //#region private methods
  private loadTreeCategory() {
    this.trees$ = this.newsDomainApiService
      .loadNewsDomain()
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
  }

  private loadCategoriesOrg(filter?: {
    page_number?: string;
    page_size?: string;
    object_type: string;
    object_name: string;
  }) {
    this.categoriesOrg$ = this.monitorObjectsService
      .loadListObjectsAdvancedSearch(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.categoriesOrg$.subscribe(data => (this.categoriesOrg = data));
  }

  private loadNewsBasket(filter?: {
    page_number?: string;
    page_size?: string;
  }) {
    this.newsBasket$ = this.newsBasketService
      .loadNewsBasketAdvancedSearch(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.newsBasket$.subscribe(data => (this.newsBasket = data));
  }

  private loadDailyNews(filter?: {
    page_number?: string;
    page_size?: string;
    title_order?: string; // or
    date_created_order?: string; // or
    rating_point_order?: string; // or
  }) {
    this.dailyNews$ = this.newsService
      .loadDailyNewsAdvancedSearch(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.dailyNews$.subscribe(data => (this.dailyNews = data));
  }

  private loadNews(
    filter?: {
      page_number?: string;
      page_size?: string;
      from_date?: string;
      to_date?: string;
      table_names: string;
      order_by?: string; // field desc/asc
    },
    data?: { field_name: string; field_value: string }[]
  ) {
    this.news$ = this.newsService.loadNewAdvancedSearch(filter, data).pipe(
      takeUntil(this.destroyed$),
      catchError(this.catchError),
      finalize(() => {
        this.isNewsLoading = true;
      })
    );
  }
  private loadListObjects(filter?: {
    page_number?: string;
    page_size?: string;
    object_type: string;
    object_name: string;
  }) {
    this.listObjects$ = this.monitorObjectsService
      .loadListObjectsAdvancedSearch(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.listObjects$.subscribe(data => (this.listObjects = data));
  }
  private catchError(err) {
    logger.debug(err);
    return of(null);
  }
  private loadNewsWebsite(filter?: {
    page_number?: string;
    page_size?: string;
    website_url?: string;
  }) {
    this.website$ = this.newsService
      .loadListWebsite(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.website$.subscribe(data => (this.wesites = data));
  }
  //#endregion
}
