import { FormControl } from '@angular/forms';
import {
  Component,
  OnInit,
  Input,
  Output,
  EventEmitter,
  OnChanges
} from '@angular/core';
import { MonitorObject } from '@vnas2-workspace/vnas2-api';
import { fadeInAnimation } from '@vnas2-workspace/vnas2-ui-commons';
import { debounceTime, tap } from 'rxjs/operators';
import { isNullOrUndefined } from 'util';

@Component({
  selector: 'app-vnas2-settings-list-item',
  templateUrl: './vnas2-settings-list-item.component.html',
  styleUrls: ['./vnas2-settings-list-item.component.scss'],
  animations: [fadeInAnimation]
})
export class Vnas2SettingsListItemComponent implements OnInit, OnChanges {
  inputSearch: FormControl = new FormControl(null);
  @Input() ObjectType: string;
  @Input() monitors: MonitorObject[] = null;
  @Output() submitForm: EventEmitter<any> = new EventEmitter<any>();
  @Output() create: EventEmitter<any> = new EventEmitter<any>();
  @Output() selectItem: EventEmitter<any> = new EventEmitter<any>();
  clickedItem: any;
  dataSource = {
    data: this.monitors
  };

  constructor() {}
  ngOnInit() {
    this.onSearchChange();
  }

  ngOnChanges() {
    this.dataSource.data = null;
    if (this.monitors) {
      setTimeout(() => {
        this.dataSource.data = this.monitors;
      }, 500);
    }
  }

  createMonitor() {
    this.clickedItem = null;
    this.create.emit({ data: true });
  }

  refreshMonitor(event) {
    event.stopPropagation();
    if (isNullOrUndefined(this.inputSearch.value)) {
      return;
    }
    this.inputSearch.setValue(null);
  }

  selectionChange(item) {
    this.clickedItem = item.object_id;
    this.selectItem.emit({ data: item });
  }
  onSearchChange(): void {
    this.inputSearch.valueChanges
      .pipe(debounceTime(1000))
      .subscribe(searchText => {
        this.submitForm.emit({ data: searchText });
      });
  }
}
