import { MonitorObject, MonitorObjectType } from '@vnas2-workspace/vnas2-api';
import { FormBuilder, FormGroup, Validators, NgForm } from '@angular/forms';
import {
  Component,
  OnInit,
  Input,
  Output,
  EventEmitter,
  OnChanges,
  ViewChild
} from '@angular/core';
import { AppConfig } from '@vnas2-workspace/vnas2-core';

@Component({
  selector: 'app-vnas2-settings-form',
  templateUrl: './vnas2-settings-form.component.html',
  styleUrls: ['./vnas2-settings-form.component.scss']
})
export class Vnas2SettingsFormComponent implements OnInit, OnChanges {
  formMonitorObject: FormGroup;
  title = 'Tạo mới';
  floatLabel = 'never';
  type: string;

  @Input() titleForm: string;
  @Input() objectTypes: MonitorObjectType[] = null;
  @Output() submitForm: EventEmitter<any> = new EventEmitter<any>();
  @Output() deleteData: EventEmitter<any> = new EventEmitter<any>();
  @ViewChild('formDirective', { static: true }) private formDirective: NgForm;
  dataSource: {
    objectTypes: MonitorObjectType[];
  } = { objectTypes: this.objectTypes };
  private object: MonitorObject = null;
  constructor(private formBuilder: FormBuilder) {}

  ngOnInit() {
    this.initializeForm();
    this.type = AppConfig.ActionType.INSERT;
  }

  ngOnChanges() {
    if (this.titleForm) {
      this.title = `Tạo mới ${this.titleForm}`;
    }
    if (this.objectTypes) {
      this.dataSource.objectTypes = this.objectTypes;
    }
  }

  onSubmit() {
    if (this.formMonitorObject.valid) {
      this.submitForm.emit({
        data: this.formMonitorObject.value,
        type: this.type
      });
    }
  }

  resetForm() {
    this.type = AppConfig.ActionType.INSERT;
    this.title = `Tạo mới ${this.titleForm}`;
    this.formMonitorObject.reset();
    this.formDirective.resetForm();
    this.object = null;
  }

  setValueForm(object?: MonitorObject) {
    if (!object) {
      return;
    }
    this.object = object;
    this.type = AppConfig.ActionType.UPDATE;
    this.title = `Cập nhật ${object.object_name}`;
    this.formMonitorObject.get('object_name').setValue(object.object_name);
    this.formMonitorObject.get('object_type').setValue(object.object_type);
    this.formMonitorObject.get('face_url').setValue(object.face_url);
    this.formMonitorObject.get('twitter_url').setValue(object.twitter_url);
    this.formMonitorObject
      .get('object_description')
      .setValue(object.object_description);
    this.formMonitorObject.get('neo4j_label').setValue(object.neo4j_label);
    this.formMonitorObject.get('image_path').setValue(object.image_path);
    this.formMonitorObject
      .get('filter_condition')
      .setValue(object.filter_condition);
    this.formMonitorObject.get('condition_str').setValue(object.condition_str);
    this.formMonitorObject.get('status').setValue(object.status);
  }

  onConfirmDelete() {
    if (!this.object) {
      return;
    }
    this.deleteData.emit({ data: this.object });
  }

  private initializeForm() {
    this.formMonitorObject = this.formBuilder.group({
      object_name: ['', [Validators.required]], // Tên tổ chức
      object_type: ['', [Validators.required]], // Loại tổ chức
      face_url: ['', []], // link facebook
      twitter_url: ['', []], // Lin]k twiter
      object_description: ['', []], // Description mô tả
      neo4j_label: ['', []], // Nhãn neo4j_label
      image_path: ['', []], // Link ảnh
      filter_condition: ['', []], // điều kiện lọc // ref mat-chips
      condition_str: ['', []], // điều kiện
      status: ['', [Validators.required]] // Thêm Checkbox Trạng thái kích hoạt (actice && diactive)
    });
  }
}
