import { Vnas2SettingsSandboxService } from './../../vnas2-settings.sandbox';
import { ActivatedRoute } from '@angular/router';
import { Component, OnInit, Injector, OnDestroy } from '@angular/core';
import {
  BaseComponent,
  Logger,
  isNotEmpty,
  AppConfig,
  SnackBarService
} from '@vnas2-workspace/vnas2-core';
import {
  PagesModel,
  IAppDialogService
} from '@vnas2-workspace/vnas2-ui-commons';
import {
  MonitorObjectsService,
  MonitorObjectDTO,
  MonitorObject
} from '@vnas2-workspace/vnas2-api';
import { Observable, Subject, of } from 'rxjs';
import { takeUntil, catchError, map, shareReplay } from 'rxjs/operators';
import { AppSettingsConfirmComponent } from '../../components/app-settings-confirm/app-settings-confirm.component';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { query } from '@angular/animations';
const logger = new Logger('Vnas2SettingsComponent');
@Component({
  selector: 'app-vnas2-settings',
  templateUrl: './vnas2-settings.component.html',
  styleUrls: ['./vnas2-settings.component.scss']
})
export class Vnas2SettingsComponent extends BaseComponent
  implements OnInit, OnDestroy {
  monitorObjects$: Observable<any>;
  monitorObjects: any;
  objectTypes$: Observable<any>;
  isMobile$: Observable<boolean> = this.breakpointObserver
    .observe([Breakpoints.XSmall, Breakpoints.Small])
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  filterMonitorObjects: {
    page_number?: string;
    page_size?: string;
    object_name?: string;
    object_type: string;
  } = {
    page_number: null,
    page_size: null,
    object_name: null,
    object_type: null
  };
  private destroyed$ = new Subject();
  constructor(
    injector: Injector,
    private activatedRoute: ActivatedRoute,
    private vnas2SettingsSandboxService: Vnas2SettingsSandboxService,
    private monitorObjectsService: MonitorObjectsService,
    private appDialogService: IAppDialogService,
    private breakpointObserver: BreakpointObserver,
    private notificationService: SnackBarService
  ) {
    super(injector);
    this.filterMonitorObjects.page_number = this.pageNumber;
    this.filterMonitorObjects.page_size = this.pageSize;
  }

  ngOnInit() {
    this.updatePageTitle();
    this.loadMonitorObjects(this.filterMonitorObjects);
    this.loadObjectTypes();
  }

  ngOnDestroy() {
    this.destroyed$.next();
    this.destroyed$.complete();
  }

  onGoToPageMonitor(page) {
    this.filterMonitorObjects.page_number = page;
    this.loadMonitorObjects(this.filterMonitorObjects);
  }

  onSearchMonitor(query?: string) {
    this.searchText(query);
  }

  onUpdateSetting(event) {
    // destructor
    const { id, data, type } = event;
    if (!data || !type) {
      return;
    }
    data.status = +data.status;
    logger.debug('onUpdateSetting', type, data);
    switch (type) {
      case AppConfig.ActionType.INSERT:
        // call func insert Settings
        this.createSettings(data).subscribe(response => {
          if (response) {
            logger.debug('create---', response);
            this.notificationService.showSuccess(`Thêm mới ${this.titlePage}`);
            this.loadMonitorObjects(this.filterMonitorObjects);
          }
        });
        break;

      case AppConfig.ActionType.UPDATE:
        // call func update Settings
        this.updateSettings(id, data).subscribe(response => {
          if (response) {
            logger.debug('update----', response);
            this.notificationService.showSuccess(
              `Cấu hình ${this.titlePage} được cập nhật!`
            );
            this.loadMonitorObjects(this.filterMonitorObjects);
          }
        });
        break;
    }
  }

  onConfirmDeleteObject(event: { data: MonitorObject }) {
    logger.debug(event);
    const { data } = event;
    const dialogRef = this.appDialogService.openConfirmYesNoDialog(
      `Xác nhận xóa ${data.object_name}`,
      AppSettingsConfirmComponent,
      AppConfig.MESSAGE.ACCEPT,
      AppConfig.MESSAGE.CANCEL,
      AppConfig.MESSAGE.CONFIRM
    );
    dialogRef.subscribe(result => {
      logger.debug(result);
      if (result) {
        // call api delete
        this.removeSettings(data.object_id).subscribe(response => {
          logger.debug(response);
          if (response) {
            this.notificationService.showSuccess(
              `Danh mục đã bị xóa khỏi cơ sở dữ liệu!`
            );
            this.loadMonitorObjects(this.filterMonitorObjects);
          }
        });
        logger.debug('// call api delete');
      }
    });
  }

  private createSettings(data: MonitorObjectDTO) {
    return this.monitorObjectsService.createMonitorObject(data).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} <br> ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private updateSettings(id: number, data: MonitorObjectDTO) {
    return this.monitorObjectsService.updateMonitorObject(id, data).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} <br> ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private removeSettings(id: number) {
    return this.monitorObjectsService.removeMonitorObject(id).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} \n ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private searchText(query: string) {
    if (isNotEmpty(query)) {
      query = query.trim();
      this.filterMonitorObjects.object_name = query;
    } else {
      this.filterMonitorObjects.object_name = null;
    }

    this.loadMonitorObjects(this.filterMonitorObjects);
  }

  private loadMonitorObjects(filter?: {
    page_number?: string;
    page_size?: string;
    object_name?: string;
    object_type: string;
  }) {
    this.monitorObjects$ = this.monitorObjectsService
      .loadListObjectsAdvancedSearch(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.monitorObjects$.subscribe(result => (this.monitorObjects = result));
  }

  private loadObjectTypes() {
    this.objectTypes$ = this.monitorObjectsService
      .loadObjectTypes()
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
  }

  private updatePageTitle() {
    const pages = this.activatedRoute.snapshot.data as PagesModel;
    this.titlePage = `${pages.title}`;
    this.filterMonitorObjects.object_type = pages.type;
    this.vnas2SettingsSandboxService.setTitle(`Cấu hình ${this.titlePage}`);
    this.setTitle(`Cấu hình ${this.titlePage}`);
  }

  private catchError(err) {
    logger.debug(err);
    return of(null);
  }
}

<div class="flex-container" [ngClass]="{ 'is-mobile': isMobile$ | async }">
  <div class="left-container">
    <app-vnas2-settings-list-item
      [ObjectType]="titlePage"
      [monitors]="monitorObjects?.data"
      (submitForm)="onSearchMonitor($event.data)"
      (create)="settingForm.resetForm()"
      (selectItem)="settingForm.setValueForm($event.data)"
    >
      <app-paging-common
        [totals]="monitorObjects?.total"
        [size]="filterMonitorObjects.page_size"
        [pageIndex]="filterMonitorObjects.page_number"
        (changePage)="onGoToPageMonitor($event.data)"
      ></app-paging-common>
    </app-vnas2-settings-list-item>
  </div>
  <div class="right-container">
    <app-vnas2-settings-form
      #settingForm
      [titleForm]="titlePage"
      [objectTypes]="(objectTypes$ | async)?.data"
      (submitForm)="onUpdateSetting($event)"
      (deleteData)="onConfirmDeleteObject($event)"
    ></app-vnas2-settings-form>
  </div>
</div>
