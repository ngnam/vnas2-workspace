import {
  Component,
  ViewEncapsulation,
  Input,
  Output,
  EventEmitter
} from '@angular/core';

@Component({
  selector: 'app-paging-common',
  templateUrl: './paging-common.component.html',
  styleUrls: ['./paging-common.component.scss'],
  encapsulation: ViewEncapsulation.ShadowDom
})
export class PagingCommonComponent {
  @Input() size: number = 20;
  @Input() totals: number = 100;
  @Input() pageIndex: number = 1;
  @Output() changePage: EventEmitter<any> = new EventEmitter<any>();

  constructor() {}

  onChangePage($event: number) {
    this.changePage.emit({
      data: $event
    });
  }
}

<div class="paging-container">
  <ngb-pagination
    [collectionSize]="totals"
    [pageSize]="size"
    [page]="pageIndex"
    ellipses="true"
    [rotate]="true"
    [maxSize]="3"
    [boundaryLinks]="false"
    (pageChange)="onChangePage($event)"
  ></ngb-pagination>
</div>
