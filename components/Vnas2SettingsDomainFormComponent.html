import {
  Component,
  EventEmitter,
  OnInit,
  Output,
  Inject,
  ViewChild
} from '@angular/core';
import { FormBuilder, FormGroup, Validators, NgForm } from '@angular/forms';
import { MAT_DIALOG_DATA } from '@angular/material';
import { NewsDomainModel } from '@vnas2-workspace/vnas2-api';

@Component({
  selector: 'app-settings-domain-form',
  templateUrl: './vnas2-settings-domain-form.component.html',
  styleUrls: ['./vnas2-settings-domain-form.component.scss']
})
export class Vnas2SettingsDomainFormComponent implements OnInit {
  floatLabel = 'never';
  title: string;
  formSettingNewsDomain: FormGroup;
  @Output() formSubmit: EventEmitter<any> = new EventEmitter();
  @ViewChild('formDirective', { static: true }) private formDirective: NgForm;
  constructor(
    private formBuilder: FormBuilder,
    @Inject(MAT_DIALOG_DATA)
    public data: { title: string; node: NewsDomainModel }
  ) {}

  ngOnInit(): void {
    this.initializedForm();
    if (this.data) {
      this.title = `${this.data.title}`;
      this.setValues(this.data.node);
    }
  }

  onSubmitForm() {
    if (this.formSettingNewsDomain.valid) {
      this.formSubmit.emit({
        data: this.formSettingNewsDomain.value
      });
      this.clearFormData();
    }
  }

  private clearFormData() {
    this.formSettingNewsDomain.reset();
    this.formDirective.resetForm();
  }

  private setValues(data?: NewsDomainModel) {
    if (!data) {
      return;
    }
    this.formSettingNewsDomain.patchValue({
      domain_name: data.domain_name,
      domain_description: data.domain_description,
      terms: data.terms,
      parent_domain_id: data.parent_domain_id
    });
  }

  private initializedForm() {
    this.formSettingNewsDomain = this.formBuilder.group({
      domain_name: ['', Validators.required],
      domain_description: ['', Validators.required],
      terms: ['', []],
      parent_domain_id: ['', []]
    });
  }
}
