import {
  Component,
  Input,
  Output,
  EventEmitter,
  OnChanges
} from '@angular/core';
import { CategoryOrganization } from '../../models/category-organization';
import { fadeInAnimation } from '../../animations';

@Component({
  selector: 'app-category-organization',
  templateUrl: './category-organization.component.html',
  styleUrls: ['./category-organization.component.scss'],
  animations: [fadeInAnimation]
})
export class CategoryOrganizationComponent implements OnChanges {
  header = {
    icon: 'vnas2-icon-earth',
    title: 'Danh mục các tổ chức'
  };

  @Input() categories: CategoryOrganization[] = null;
  @Output() clickOrganization: EventEmitter<any> = new EventEmitter<any>();

  dataSouce = {
    data: this.categories
  };

  constructor() {}

  ngOnChanges() {
    this.dataSouce.data = null;
    if (this.categories) {
      setTimeout(() => {
        this.dataSouce.data = this.categories;
      }, 500);
    }
  }

  onClickOrganization($event: CategoryOrganization) {
    this.clickOrganization.emit({ organization: $event });
  }
}
