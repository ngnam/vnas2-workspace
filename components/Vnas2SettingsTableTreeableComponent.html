import {
  Component,
  OnInit,
  Input,
  Output,
  EventEmitter,
  OnDestroy,
  OnChanges
} from '@angular/core';
import { AppConfig } from '@vnas2-workspace/vnas2-core';
import { FormControl } from '@angular/forms';
import { isNullOrUndefined } from 'util';
import { debounceTime } from 'rxjs/operators';
import { NewsDomainModel, NewsDomainDTO } from '@vnas2-workspace/vnas2-api';
import { BehaviorSubject } from 'rxjs';
import { MatTreeNestedDataSource } from '@angular/material';
import { NestedTreeControl } from '@angular/cdk/tree';
import {
  IAppDialogService,
  fadeInAnimation
} from '@vnas2-workspace/vnas2-ui-commons';
import { Vnas2SettingsDomainFormComponent } from '../vnas2-settings-domain-form/vnas2-settings-domain-form.component';
import { Vnas2SettingsDomainConfirmDeleteComponent } from '../vnas2-settings-domain-confirm-delete/vnas2-settings-domain-confirm-delete.component';

@Component({
  selector: 'app-vnas2-settings-table-treeable',
  templateUrl: './vnas2-settings-table-treeable.component.html',
  styleUrls: ['./vnas2-settings-table-treeable.component.scss'],
  animations: [fadeInAnimation]
})
export class Vnas2SettingsTableTreeableComponent
  implements OnInit, OnDestroy, OnChanges {
  /** dummy for getting unique ids */
  private static _id = 0;
  inputSearch: FormControl = new FormControl(null);
  @Input() ObjectType: string;
  @Input() NewsDomain: NewsDomainModel[] = null;
  @Output() createNode: EventEmitter<any> = new EventEmitter<any>();
  @Output() updateNode: EventEmitter<any> = new EventEmitter<any>();
  @Output() deleteNode: EventEmitter<any> = new EventEmitter<any>();
  @Output() childrensNode: EventEmitter<any> = new EventEmitter<any>();
  isClicked = false;

  /** tree source stuff */
  readonly dataSource$: BehaviorSubject<NewsDomainModel[]>;
  readonly treeSource: MatTreeNestedDataSource<NewsDomainModel>;
  /** tree control */
  readonly treeControl = new NestedTreeControl<NewsDomainModel>(
    node => node.children
  );
  readonly hasChild = (_: number, node: NewsDomainModel) =>
    !!node.children && node.children.length > 0;
  readonly trackBy = (_: number, node: NewsDomainModel) => node.domain_id;

  constructor(private appDialogService: IAppDialogService) {
    this.treeSource = new MatTreeNestedDataSource<NewsDomainModel>();
    // this dataSource is not required but its rly. helpfull to think reactive
    this.dataSource$ = new BehaviorSubject<NewsDomainModel[]>([]);
    this.dataSource$.subscribe(items => {
      this.treeSource.data = null;
      this.treeSource.data = items;
    });
  }

  ngOnInit() {
    this.onSearchChange();
  }

  ngOnChanges() {
    if (this.NewsDomain) {
      this.initData(this.NewsDomain);
    }
  }

  addNodeToTree(node?: NewsDomainModel, children?: NewsDomainModel[]) {
    // add as child
    if (node) {
      node.children = [...(node.children || []), ...children];
      if (!this.treeControl.isExpanded(node)) {
        this.treeControl.expand(node);
      }
    } else {
      this.dataSource$.next([...this.dataSource$.value, ...children]);
    }
    this.dataSource$.next(this.dataSource$.value);
  }

  // mapping nodeDTO -> node current
  mappingNode(node?: NewsDomainModel, nodeDTO?: NewsDomainDTO) {
    node.domain_description = nodeDTO.domain_description;
    node.domain_name = nodeDTO.domain_name;
    node.parent_domain_id = nodeDTO.parent_domain_id;
    node.terms = nodeDTO.terms;
  }

  /** toggle node */
  toggleNode(node: NewsDomainModel) {
    this.treeControl.toggle(node);
  }

  openDialog(node?: NewsDomainModel, type?: string) {
    const newsModel = Object.assign({}, node);
    let title = node ? newsModel.domain_name : '';

    switch (type) {
      case AppConfig.ActionType.INSERT:
        title = `Thm danh mc tin trong danh mc: ${title}/{Danh mc mi}`;
        newsModel.domain_description = null;
        newsModel.domain_name = null;
        newsModel.terms = null;
        newsModel.parent_domain_id = newsModel.domain_id;
        break;
      case AppConfig.ActionType.UPDATE:
        title = `Sa danh mc tin: ${title}`;
        break;
    }

    const data = {
      title,
      node: newsModel
    };
    const dialogRef = this.appDialogService.openDialogCustom(
      data,
      Vnas2SettingsDomainFormComponent
    );
    const subFormSubmit = dialogRef.componentInstance.formSubmit.subscribe(
      result => {
        if (result) {
          if (type === AppConfig.ActionType.INSERT) {
            this.createNode.emit({ node, result });
            // add node new to node
          }
          if (type === AppConfig.ActionType.UPDATE) {
            this.updateNode.emit({ id: newsModel.domain_id, node, result });
            // update node edit
          }
          setTimeout(() => {
            dialogRef.close();
          }, 200);
        }
      }
    );

    dialogRef.afterClosed().subscribe(() => {
      subFormSubmit.unsubscribe();
    });
  }

  openConfirmDialog(node?: NewsDomainModel, nodeDom?: any) {
    const newsDomain = Object.assign({}, node);
    const title = node ? `Xc nhn xa danh mc ${newsDomain.domain_name}` : '';
    const data: {
      title: string;
      node?: NewsDomainModel;
      buttonYesText?: string;
      buttonNoText?: string;
      messageConfirm?: string;
      isConfirm: true;
    } = {
      title,
      node: newsDomain,
      buttonNoText: AppConfig.MESSAGE.CANCEL,
      buttonYesText: AppConfig.MESSAGE.ACCEPT,
      messageConfirm: AppConfig.MESSAGE.CONFIRM,
      isConfirm: true
    };
    const dialogRef = this.appDialogService.openDialogConfirmCustom(
      data,
      Vnas2SettingsDomainConfirmDeleteComponent
    );
    dialogRef.subscribe(result => {
      if (result) {
        this.deleteNode.emit({ data: newsDomain.domain_id, nodeDom });
      }
    });
  }

  openDialogCreateNodeRoot() {
    const newsModel: NewsDomainDTO = {
      domain_description: null,
      domain_name: null,
      parent_domain_id: null,
      terms: null
    };
    const title = `Thm mi danh mc tin tc (ROOT TREE)`;

    const data = {
      title,
      node: newsModel
    };
    const dialogRef = this.appDialogService.openDialogCustom(
      data,
      Vnas2SettingsDomainFormComponent
    );
    const subFormSubmit = dialogRef.componentInstance.formSubmit.subscribe(
      result => {
        if (result) {
          this.createNode.emit({ result });
          setTimeout(() => {
            dialogRef.close();
          }, 200);
        }
      }
    );
    dialogRef.afterClosed().subscribe(() => {
      subFormSubmit.unsubscribe();
    });
  }

  getChildrenNode(node?: NewsDomainModel, tmpButton?: any) {
    this.childrensNode.emit({ data: node, button: tmpButton });
  }

  /** init tree data */
  private initData(nodes) {
    this.dataSource$.next(nodes);
  }

  /** creates a new tree item */
  private _createTreeItem(
    domainName: string,
    ...children: NewsDomainModel[]
  ): NewsDomainModel {
    return {
      domain_id: ++Vnas2SettingsTableTreeableComponent._id,
      domain_name: domainName,
      children
    };
  }

  ngOnDestroy() {
    this.dataSource$.next([]);
    this.dataSource$.complete();
  }

  onRefresh(event) {
    event.stopPropagation();
    if (isNullOrUndefined(this.inputSearch.value)) {
      return;
    }
    this.inputSearch.setValue(null);
  }

  private onSearchChange(): void {
    this.inputSearch.valueChanges
      .pipe(debounceTime(1000))
      .subscribe(searchText => {
        //  this.searchForm.emit({ data: searchText });
        // this.DataTable.filter = searchText;
      });
  }
}
