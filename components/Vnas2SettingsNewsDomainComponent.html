import {
  Component,
  Injector,
  OnInit,
  OnDestroy,
  ViewChild,
  Renderer2
} from '@angular/core';
import { PagesModel } from '@vnas2-workspace/vnas2-ui-commons';
import {
  NewsDomainApiService,
  NewsDomainDTO
} from '@vnas2-workspace/vnas2-api';

import { ActivatedRoute } from '@angular/router';
import {
  BaseComponent,
  Logger,
  SnackBarService,
  AppConfig
} from '@vnas2-workspace/vnas2-core';
import { Vnas2SettingsSandboxService } from '../../vnas2-settings.sandbox';
import { Observable, Subject, of } from 'rxjs';
import { takeUntil, catchError } from 'rxjs/operators';
import { Vnas2SettingsTableTreeableComponent } from '../../components/vnas2-settings-table-treeable/vnas2-settings-table-treeable.component';
import { MatButton } from '@angular/material';

const logger = new Logger('Vnas2SettingsNewsDomainComponent');

@Component({
  selector: 'app-vnas2-settings-news-domain',
  templateUrl: './vnas2-settings-news-domain.component.html',
  styleUrls: ['./vnas2-settings-news-domain.component.scss']
})
export class Vnas2SettingsNewsDomainComponent extends BaseComponent
  implements OnInit, OnDestroy {
  objectType: string;

  treeDataNewsDomain$: Observable<any>;
  treeDataNewsDomain: any;

  filterTreeNewsDomain: {
    page_number?: string;
    page_size?: string;
    parent_id?: string;
  } = {
    page_number: null,
    page_size: null,
    parent_id: null
  };

  private destroyed$ = new Subject();
  @ViewChild('treeTable', { static: true })
  treeTable: Vnas2SettingsTableTreeableComponent;
  constructor(
    injector: Injector,
    private renderer2: Renderer2,
    private activatedRoute: ActivatedRoute,
    private vnas2SettingsSandboxService: Vnas2SettingsSandboxService,
    private newsDomainApiService: NewsDomainApiService,
    private notificationService: SnackBarService
  ) {
    super(injector);
    this.filterTreeNewsDomain.page_number = this.pageNumber;
    this.filterTreeNewsDomain.page_size = this.pageSize;
  }

  ngOnInit() {
    this.updatePageTitle();
    this.loadNewsDomainsTree(this.filterTreeNewsDomain);
  }

  ngOnDestroy() {
    this.destroyed$.next();
    this.destroyed$.complete();
  }

  onCreateTree(event: any) {
    const { result, node } = event;
    logger.debug('onCreateTree: ', result, node);

    logger.debug(result, node);
    if (result && result.data) {
      this.createNewsDoamin(result.data).subscribe(response => {
        if (response && response.data) {
          logger.debug('create---', response.data);
          this.notificationService.showSuccess(
            `Thêm mới danh mục ${response.data.domain_name}`
          );
          this.treeTable.addNodeToTree(node, [response.data]);
        }
      });
    }
  }

  onUpdateTree(event: any) {
    const { id, result, node } = event;
    logger.debug(id, result, node);
    if (!(result && result.data)) {
      return;
    }
    this.updateNewsDomain(id, result.data).subscribe(response => {
      if (response) {
        logger.debug('update----', response);
        this.notificationService.showSuccess(
          `Danh mục ${result.data.domain_name} được cập nhật!`
        );
        // update node in treeable
        this.treeTable.mappingNode(node, result.data);
      }
    });
  }

  onDeleteTree(event: any) {
    logger.debug('onDeleteTree: ', event);
    const { data, nodeDom } = event;
    this.removeNewsDomain(data).subscribe(response => {
      logger.debug(response);
      if (response) {
        this.notificationService.showSuccess(
          `Danh mục đã bị xóa khỏi cơ sở dữ liệu!`
        );
        if (nodeDom && nodeDom.parentNode) {
          nodeDom.parentNode.remove();
        }
      }
    });
  }

  onGetChildrensNode(event: any) {
    const { data, button } = event;
    if (!data) {
      return;
    }
    this.loadSubNewsDomainsTree(data.domain_id).subscribe(result => {
      logger.debug('loadSubNewsDomainsTree', result);
      if (!result) {
        return;
      }
      if (result.data && result.data.length === 0) {
        logger.debug('The Node is not the childrens');
        this.notificationService.showInfo(
          'Danh mục này không có danh mục con!'
        );
        this.hideElement(button);
        return;
      }
      this.hideElement(button);
      this.treeTable.addNodeToTree(data, result.data);
    });
  }

  onGoToPageMonitor(page) {
    this.filterTreeNewsDomain.page_number = page;
    this.loadNewsDomainsTree(this.filterTreeNewsDomain);
  }

  private loadNewsDomainsTree(filter?: {
    page_number?: string;
    page_size?: string;
    parent_id?: string;
  }) {
    this.treeDataNewsDomain$ = this.newsDomainApiService
      .loadTrees(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
    this.treeDataNewsDomain$.subscribe(result => {
      if (!result) {
        return;
      }
      this.treeDataNewsDomain = result;
    });
  }

  private hideElement(element) {
    if (element) {
      logger.debug(element);
      if (element instanceof MatButton) {
        setTimeout(() => {
          element._elementRef.nativeElement.remove();
        }, 500);

        logger.debug('removed', element._elementRef.nativeElement);
      }
    }
  }

  private loadSubNewsDomainsTree(domainId: string) {
    const filter: {
      page_number?: string;
      page_size?: string;
      parent_id?: string;
    } = {
      page_number: this.pageNumber,
      page_size: this.pageSize,
      parent_id: domainId
    };
    return this.newsDomainApiService
      .loadTrees(filter)
      .pipe(takeUntil(this.destroyed$), catchError(this.catchError));
  }

  private createNewsDoamin(data: NewsDomainDTO) {
    return this.newsDomainApiService.addNewsDomain(data).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} <br> ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private updateNewsDomain(id: string, data: NewsDomainDTO) {
    return this.newsDomainApiService.updateNewsDomain(id, data).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} \n ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private removeNewsDomain(id: string) {
    return this.newsDomainApiService.removeNewsDomain(id).pipe(
      takeUntil(this.destroyed$),
      catchError((err: any) => {
        const { error } = err;
        this.notificationService.showError(
          `${AppConfig.MESSAGE.ERROR_SERVER} \n ${error!.detail}`
        );
        return this.catchError(err);
      })
    );
  }

  private updatePageTitle() {
    const pages = this.activatedRoute.snapshot.data as PagesModel;
    this.titlePage = `${pages.title}`;
    this.vnas2SettingsSandboxService.setTitle(`Cấu hình ${this.titlePage}`);
    this.setTitle(`Cấu hình ${this.titlePage}`);
  }

  private catchError(err) {
    logger.debug(err);
    return of(null);
  }
}

<div class="flex-container">
  <div class="left-container">
