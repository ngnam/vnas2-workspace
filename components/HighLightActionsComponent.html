import { Component, OnInit, Input, OnChanges } from '@angular/core';
export interface Actions {
  content: string;
  date: Date;
}
@Component({
  selector: 'app-high-light-actions',
  templateUrl: './high-light-actions.component.html',
  styleUrls: ['./high-light-actions.component.scss']
})
export class HighLightActionsComponent implements OnChanges {
  @Input() data: Actions[] = null;

  dataSource = {
    data: this.data
  };

  constructor() {}

  ngOnChanges() {
    this.dataSource.data = null;
    if (this.data) {
      setTimeout(() => {
        this.dataSource.data = this.data;
      }, 600);
    }
  }
}
