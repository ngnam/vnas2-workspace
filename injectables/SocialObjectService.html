import { SocialObjectDTO } from './../models/social-object';
import { mapToHttpParamsQuery, fmt } from '@vnas2-workspace/vnas2-core';
import { Injectable } from '@angular/core';
import { ApiDataService } from './api-service';
const router = {
  social_object: `/api/social_object`,
  create_social_object: `/api/social_object`,
  update_social_object: `/api/social_object/{socialObjectId}`,
  delete_social_object: `/api/social_object/{socialObjectId}`
};
@Injectable()
export class SocialObjectService {
  constructor(private httpClient: ApiDataService) {}

  loadSocialObjectAdvancedSearch(filter?: {
    page_size?: string;
    page_number?: string;
  }) {
    const params = mapToHttpParamsQuery(filter);
    return this.httpClient.get(router.social_object, params);
  }

  createSocialObject(data: SocialObjectDTO) {
    return this.httpClient.post(router.create_social_object, data);
  }

  updateSocialObject(socialObjectId: number, data: SocialObjectDTO) {
    const uri = fmt(router.update_social_object, { socialObjectId });
    return this.httpClient.put(uri, data);
  }

  removeSocialObject(socialObjectId: number) {
    const uri = fmt(router.delete_social_object, { socialObjectId });
    return this.httpClient.delete(uri);
  }
}
