import { CommonModule } from '@angular/common';
import { ReactiveFormsModule } from '@angular/forms';
import { NgModule, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
import { NgbPaginationModule } from '@ng-bootstrap/ng-bootstrap';
import {
  DateAdapter,
  MAT_DATE_FORMATS,
  MAT_DATE_LOCALE
} from '@angular/material';
import { APP_DATE_FORMATS } from './adapters/app-date-format';
import { AppDateAdapter } from './adapters/app-date.adapter';
import { MaterialModule } from './material/material.module';

import { SortByPipe } from './pipes/sortBy.pipe';
import { CountDatePipe } from './pipes/count-date.pipe';

import { CategoryOrganizationComponent } from './components/category-organization/category-organization.component';
import { ListObjectsComponent } from './components/list-objects/list-objects.component';
import { NewsBasketComponent } from './components/news-basket/news-basket.component';
import { PagingCommonComponent } from './components/paging-common/paging-common.component';
import { TreeCategoryVerticalComponent } from './components/tree-category-vertical/tree-category-vertical.component';
import { DefaultImageDirective } from './directives/default-image.directive';
import { NewsWithDomainComponent } from './components/news-with-domain/news-with-domain.component';
import { FormSearchNewsAdvancedComponent } from './components/form-search-news-advanced/form-search-news-advanced.component';
import { DailyNewsComponent } from './components/daily-news/daily-news.component';
import { NewsSummaryComponent } from './components/news-summary/news-summary.component';
import { HighLightActionsComponent } from './components/high-light-actions/high-light-actions.component';
import { RenderDebugDirective } from './directives/render.debug.directive';

const DIRECTIVES = [DefaultImageDirective, RenderDebugDirective];

const COMPONENTS = [
  TreeCategoryVerticalComponent,
  CategoryOrganizationComponent,
  NewsBasketComponent,
  DailyNewsComponent,
  ListObjectsComponent,
  NewsWithDomainComponent,
  FormSearchNewsAdvancedComponent,
  NewsSummaryComponent,
  HighLightActionsComponent,
  PagingCommonComponent
];

const PIPES = [SortByPipe, CountDatePipe];

@NgModule({
  declarations: [...COMPONENTS, ...DIRECTIVES, ...PIPES],
  imports: [
    CommonModule,
    MaterialModule,
    ReactiveFormsModule,
    NgbPaginationModule
  ],
  exports: [...COMPONENTS, ...DIRECTIVES, ...PIPES],
  providers: [
    { provide: DateAdapter, useClass: AppDateAdapter },
    { provide: MAT_DATE_FORMATS, useValue: APP_DATE_FORMATS },
    { provide: MAT_DATE_LOCALE, useValue: 'vn-VI' }
  ],
  schemas: [CUSTOM_ELEMENTS_SCHEMA]
})
export class Vnas2UiCommonsModule {}
